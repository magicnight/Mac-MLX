Replace the body of `SpeedSampler` with:

```swift
internal final class SpeedSampler: @unchecked Sendable {
    /// Lower alpha = smoother. 0.15 still reacts to a real slowdown
    /// within ~6 samples but masks single-callback spikes.
    private let alpha = 0.15
    /// Minimum elapsed time between EMA updates. URLSession fires
    /// progress every few ms during LFS downloads; we only need a
    /// ~2 Hz cadence for a stable ETA display.
    private let minSampleInterval: TimeInterval = 0.5

    private let lock = NSLock()
    private var windowStart: Date?
    private var windowStartBytes: Int64 = 0
    private var ema: Double = 0

    func record(bytes: Int64) -> Double {
        lock.lock(); defer { lock.unlock() }
        let now = Date()
        guard let start = windowStart else {
            windowStart = now
            windowStartBytes = bytes
            return 0
        }
        let dt = now.timeIntervalSince(start)
        guard dt >= minSampleInterval else { return ema }
        let dbytes = Double(bytes - windowStartBytes)
        guard dbytes >= 0 else { return ema }
        let instantaneous = dbytes / dt
        ema = ema == 0
            ? instantaneous
            : alpha * instantaneous + (1 - alpha) * ema
        windowStart = now
        windowStartBytes = bytes
        return ema
    }
}
```

- [ ] **Step 1: Add a post-filter helper**

```swift
/// HF's `search=` param is too lenient — "gemma-4" returns gemma-3,
/// gemma-2, etc. Post-filter so the visible query tokens must all
/// appear in the model's repo name (case-insensitive). Preserves
/// HF-side download-count ordering.
private static func tokenizeQuery(_ query: String) -> [String] {
    query
        .split(whereSeparator: { !$0.isLetter && !$0.isNumber })
        .map { $0.lowercased() }
        .filter { !$0.isEmpty }
}

private static func matches(_ model: HFModel, queryTokens: [String]) -> Bool {
    let name = (model.id.split(separator: "/").last.map(String.init) ?? model.id).lowercased()
    return queryTokens.allSatisfy { name.contains($0) }
}
```

- [ ] **Step 1: Define the MessageSegment enum**

`macMLX/macMLX/Views/Chat/MessageSegment.swift`:

```swift
import Foundation

enum MessageSegment: Equatable {
    case text(String)
    /// `isClosed == false` ⇒ streaming mid-think, animation must run.
    case think(String, isClosed: Bool)
}

enum MessageSegmenter {
    /// Parse message content into text + think segments. Handles:
    /// - Normal `<think>…</think>X` (produces think + text).
    /// - Streaming `<think>partial` (produces open think).
    /// - Missing opener `…</think>X` — treat everything before
    ///   `</think>` as think (fixes qwen3's template-injected opener).
    /// - No think tags at all — single `.text` segment.
    static func parse(_ content: String) -> [MessageSegment] {
        var segments: [MessageSegment] = []
        var remaining = content[...]
        let openTag = "<think>"
        let closeTag = "</think>"

        // Handle the implicit-open case ONCE up-front.
        if !remaining.contains(openTag), let closeRange = remaining.range(of: closeTag) {
            let think = String(remaining[remaining.startIndex..<closeRange.lowerBound])
            if !think.isEmpty { segments.append(.think(think, isClosed: true)) }
            remaining = remaining[closeRange.upperBound...]
        }

        while let openRange = remaining.range(of: openTag) {
            let before = String(remaining[remaining.startIndex..<openRange.lowerBound])
            if !before.isEmpty { segments.append(.text(before)) }
            let afterOpen = remaining[openRange.upperBound...]
            if let closeRange = afterOpen.range(of: closeTag) {
                let think = String(afterOpen[afterOpen.startIndex..<closeRange.lowerBound])
                segments.append(.think(think, isClosed: true))
                remaining = afterOpen[closeRange.upperBound...]
            } else {
                segments.append(.think(String(afterOpen), isClosed: false))
                remaining = afterOpen[afterOpen.endIndex...]
                break
            }
        }
        let trailing = String(remaining)
        if !trailing.isEmpty { segments.append(.text(trailing)) }
        return segments
    }
}
```
